create database pizzaria_ze;

use pizzaria_ze;

CREATE SCHEMA [cadastro];

CREATE SCHEMA [pedido];

CREATE TABLE cadastro.tb_funcionario (
    id INT PRIMARY KEY Identity,
    nome VARCHAR(45),
    cpf VARCHAR(11),
    matricula VARCHAR(15),
    senha VARCHAR(256),
    grupo CHAR(1),
    possuiHabilitacao bit,
    cnh VARCHAR(15),
    validade_motorista DATE,
    observacao VARCHAR(45),
    telefone VARCHAR(11),
    email VARCHAR(45),
    logradouro VARCHAR(100),
    bairro varchar (50),
    numero VARCHAR(10),
	cep VARCHAR(8),
    cidade varchar (50),
    pais varchar (50),
    complemento VARCHAR(45),
    excluido bit,
);

CREATE TABLE cadastro.tb_cliente (
    id INT PRIMARY KEY Identity,
    nome VARCHAR(50),
    cpf VARCHAR(11) UNIQUE,
    telefone VARCHAR(11),
    email VARCHAR(45),
    logradouro VARCHAR(100),
    bairro varchar (50),
    numero VARCHAR(10),
	cep VARCHAR(8),
    cidade varchar (50),
    pais varchar (50),
    complemento VARCHAR(45),
    excluido bit,
);

CREATE TABLE cadastro.tb_ingrediente (
    id INT PRIMARY KEY Identity,
    nome VARCHAR(100) UNIQUE
);

CREATE TABLE cadastro.tb_sabor (
    id INT PRIMARY KEY Identity,
    nome VARCHAR(50) UNIQUE,
    foto Varbinary(MAX),
    categoria CHAR(1),
    tipo CHAR(1)
);

CREATE TABLE cadastro.ingredientes_sabores (
    id_ingrediente_sabor INT PRIMARY KEY Identity,
    ingrediente_id INT,
    sabor_id INT
);

CREATE TABLE cadastro.tb_produto (
    id INT PRIMARY KEY Identity,
    nome VARCHAR(45),
    valor DECIMAL(11,2),
    tipo CHAR(1),
    medida_unitaria VARCHAR(20)
);

CREATE TABLE cadastro.tb_valor (
    id INT PRIMARY KEY Identity,
    tamanho char(1),
    tradicional DECIMAL(11,2),
    especial DECIMAL(11,2),
    bordaTradicional DECIMAL(11,2),
    bordaEspecial DECIMAL(11,2),
);

ALTER TABLE cadastro.ingredientes_sabores ADD CONSTRAINT FK_ingredientes_sabores
    FOREIGN KEY (ingrediente_id)
    REFERENCES cadastro.tb_ingrediente (id);
 
ALTER TABLE cadastro.ingredientes_sabores ADD CONSTRAINT FK_ingrediente_sabor
    FOREIGN KEY (sabor_id)
    REFERENCES cadastro.tb_sabor (id);

CREATE TABLE pedido.tb_pedido (
    id INT PRIMARY KEY Identity,
    dataPedido DATE,
    ClienteId INT,
    FuncionarioId INT,
);

CREATE TABLE pedido.tb_pizza (
    id INT PRIMARY KEY Identity,
    PedidoId INT,
    tamanho char(1),
    tipoBorda CHAR(1),
    valor DECIMAL(11,2),
);


CREATE TABLE pedido.pizza_sabor(
    PizzaId INT,
    SaborId INT
);

ALTER TABLE pedido.pizza_sabor ADD CONSTRAINT FK_pizzas_sabores
    FOREIGN KEY (PizzaId)
    REFERENCES pedido.tb_pizza (id);

ALTER TABLE pedido.pizza_sabor ADD CONSTRAINT FK_sabores_pizzas
    FOREIGN KEY (SaborId)
    REFERENCES cadastro.tb_sabor (id);


CREATE TABLE pedido.pedido_produto(
    PedidoId INT,
    ProdutoId INT
);


ALTER TABLE pedido.pedido_produto ADD CONSTRAINT FK_pedidos_produtos
    FOREIGN KEY (PedidoId)
    REFERENCES pedido.tb_pedido (id);

ALTER TABLE pedido.pedido_produto ADD CONSTRAINT FK_produtos_pedidos
    FOREIGN KEY (ProdutoId)
    REFERENCES cadastro.tb_produto (id);



ALTER TABLE pedido.pizza_sabor ADD CONSTRAINT FK_sabores_pizzas
    FOREIGN KEY (SaborId)
    REFERENCES cadastro.tb_sabor (id);

ALTER TABLE pedido.pedido_produto ADD CONSTRAINT FK_produtos_pedidos
    FOREIGN KEY (ProdutoId)
    REFERENCES cadastro.tb_produto (id);
